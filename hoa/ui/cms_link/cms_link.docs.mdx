import { CMSLink } from "./cms_link"
CMSLink.performAction = action => {
  alert(`received action: ${action}`)
}
CMSLink.getInternalLink = document => [
  props => <a {...props} />,
  { href: document.path.current, children: "Internal Link" },
]

## Configuration

You will need to configure the `getInternalLink` function, which is used to generate internal
links in your project. Additionally, you'll need to configure `performAction` if you want to
support action links. Here is an example configuration for use with Gatsby:

```js
import { Link } from "gatsby"
import { CMSLink } from "@hoa/hoa.ui.cms_link"

CMSLink.performAction = action => {
  alert(`received action: ${action}`)
}

CMSLink.getInternalLink = document => [Link, { to: document.path.current }]
```

## Component usage

See compositions for examples of styling links and using render props.

### Action Link

```js live
<CMSLink
  link={{
    _key: "e2d5a7edc3e6",
    _type: "actionLink",
    action: "bookingDialog",
  }}
>
  Action Link
</CMSLink>
```

### External Link

```js live
<CMSLink link={{ _type: "externalLink", url: "http://example.com" }}>
  External Link
</CMSLink>
```

### Internal Link

```js live
<CMSLink
  link={{
    _key: "d3d49d559627",
    _type: "internalLink",
    document: {
      _type: "collection",
      path: { _type: "slug", current: "/apparel" },
    },
  }}
>
  Internal Link
</CMSLink>
```

```js live
<CMSLink
  link={{
    _type: "videoModalLink",
    video: {
      _type: "mux.videoAsset",
      data: { aspect_ratio: "1024:473" },
      playbackId: "GM01ihvcrnrjUx02dcVYe26vLDWuWUN00f9dhOGo6Pg00vY",
      thumbTime: 4.621051,
    },
  }}
>
  Video Link
</CMSLink>
```

### Optional Link

```js live
<CMSLink isOptional>
  A <code>CMSLink</code> with no link is just a <code>span</code>
</CMSLink>
```
