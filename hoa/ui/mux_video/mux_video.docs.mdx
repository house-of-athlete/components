Renders a video player for a video hosted by Mux.

```js live
<MuxVideo
  style={{ height: "auto", width: "100%" }}
  autoPlay
  muted
  loop
  playsInline
  video={[
    {
      video: {
        _type: "mux.videoAsset",
        data: { aspect_ratio: "4:5" },
        playbackId: "xxpHrEQMIRfCMkKfD1xJ7yDeUjvlq00NzpUZWa02EgLIs",
        thumbTime: null,
      },
      media: "(max-width: 769px)",
    },
    {
      video: {
        _type: "mux.videoAsset",
        data: { aspect_ratio: "1024:473" },
        playbackId: "GM01ihvcrnrjUx02dcVYe26vLDWuWUN00f9dhOGo6Pg00vY",
        thumbTime: 4.621051,
      },
      media: "(min-width: 770px)",
    },
  ]}
/>
```

## Player with controls

```js live
<MuxVideoPlayer
  playsInline
  video={{
    _type: "mux.videoAsset",
    data: { aspect_ratio: "1024:473" },
    playbackId: "GM01ihvcrnrjUx02dcVYe26vLDWuWUN00f9dhOGo6Pg00vY",
    thumbTime: 4.621051,
  }}
>
  <div className="absolute flex inset-0 items-center justify-center pointer-events-none">
    Play Button!
  </div>
</MuxVideoPlayer>
```
